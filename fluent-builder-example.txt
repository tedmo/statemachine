new StateMachineModelBuilder()
    .states(INIT, SUSPENDED, ACTIVE)
    .initialState(INIT)
    .transition()
        .from(INIT)
        .to(SUSPENDED)
        .on(CreatEvent.class)
        .when(suspendedStatusCondition)
    .transition()
        .from(INIT)
        .to(ACTIVE)
        .on(CreateEvent.class)
        .when(activeStateCondition)
    .transition()
        .from(SUSPENDED)
        .to(ACTIVE)
        .on(UpdateEvent.class)
        .or(OtherUpdateEvent.class) ???
        .when(activeStatusCondition)
    .transition()
        .fromAnyState()
        .to(CANCELLED)
        .on(CancelEvent.class)
        .when(cancelCondition)
    .transition()
        .from(ACTIVE)
        .to(CANCELLED)
        .onAnyEvent()
        .withoutCondition() ??? always() ???
    .action()
        .in(INIT)
        .on(CreateEvent.class)
        .doAction(createSODCAction)
    .action()
        .entering(SUSPENDED)
        .onAnyEvent()
        .doAction(logSuspendedOnEnterAction)
    .action()
        .enteringAnyState()
        .onAnyEvent()
        .doAction(persistAction)
